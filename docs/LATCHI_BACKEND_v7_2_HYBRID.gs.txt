function json_(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
function ok_(obj) { return json_(Object.assign({status:'success', success:true}, obj || {})); }
function err_(msg, code) { return json_({status:'error', success:false, code:code||400, message:String(msg||'error')}); }

function getConfig_() {
  const props = PropertiesService.getScriptProperties();
  const raw = props.getProperty(CONFIG_KEY);
  if (!raw) {
    const def = getDefaultConfig_();
    props.setProperty(CONFIG_KEY, JSON.stringify(def));
    return def;
  }
  try { return Object.assign({}, getDefaultConfig_(), JSON.parse(raw)); }
  catch(e) { const def=getDefaultConfig_(); props.setProperty(CONFIG_KEY, JSON.stringify(def)); return def; }
}
function saveConfig_(c) { PropertiesService.getScriptProperties().setProperty(CONFIG_KEY, JSON.stringify(c)); }
function bumpRevision_(c) { c.server_revision = (Number(c.server_revision)||1)+1; return c.server_revision; }

function isAdmin_(p) {
  return String(p.password||'') === ADMIN_PASSWORD || String(p.secret||'') === LEGACY_SECRET;
}
function readBody_(e) {
  if (e && e.postData && e.postData.contents) {
    try { return JSON.parse(e.postData.contents); } catch(_) {}
  }
  return (e && e.parameter) ? e.parameter : {};
}

function doGet(e) {
  try {
    const p = (e && e.parameter) ? e.parameter : {};
    const action = String(p.action || '').trim();
    // توافق Android القديم: بدون action لكن فيه code/device_id
    if (!action && p.code) return handleVerifyCode_(p, true);
    switch(action) {
      case 'get_config': return ok_(getPublicConfig_());
      case 'get_master_config': return handleGetMasterConfig_();
      case 'get_live_master_state': return handleGetLiveMasterState_();
      case 'verify_code': return handleVerifyCode_(p, false);
      case 'validate_code': return handleVerifyCode_(p, true);
      case 'get_categories': return handleGetCategories_(p);
      case 'get_app_update': case 'check_app_update': return handleGetAppUpdate_(p);
      case 'ping': return handlePing_(p);
      case 'get_sessions': return handleGetSessions_(p);
      case 'inspect_source': return handleInspectSource_(p);
      // توافق لوحة التحكم القديمة عبر GET + secret
      case 'update_master_url': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminUpdateMasterUrl_(p.master_url || p.playlist_url || p.url || '');
      case 'toggle_app_mode': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminToggleAppMode_(p.mode || 'free');
      case 'force_revision': case 'increment_server_revision': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminForceRevision_();
      case 'set_app_update': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminSetAppUpdate_(p);
      case 'save_category_overrides': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminSaveCategoryOverrides_(legacyOverridesFromParams_(p));
      case 'save_view_config': case 'set_hidden_categories': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminSetHiddenCategories_(parseHidden_(p.hidden_categories || p.hidden || ''));
      case 'add_code': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminAddVipCode_(p);
      case 'edit_user': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminEditVipCode_(p);
      case 'delete_user': case 'delete_vip_code': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminDeleteVipCode_(p.code);
      case 'get_vip_codes': case 'get_all_users': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminGetVipCodes_();
      case 'clear_vip_codes': case 'delete_all_users': if (!isAdmin_(p)) return err_('Unauthorized',403); return adminClearVipCodes_();
      default: return err_('Unknown action: '+action);
    }
  } catch(ex) { return err_(ex.toString()); }
}

function doPost(e) {
  try {
    const body = readBody_(e);
    if (!isAdmin_(body)) return err_('Unauthorized: wrong password',403);
    const action = String(body.action || '').trim();
    switch(action) {
      case 'update_master_url': return adminUpdateMasterUrl_(body.master_url || body.playlist_url || body.url || '');
      case 'toggle_app_mode': return adminToggleAppMode_(body.mode);
      case 'force_revision': case 'increment_server_revision': return adminForceRevision_();
      case 'set_app_update': return adminSetAppUpdate_(body);
      case 'add_vip_code': case 'add_code': return adminAddVipCode_(body);
      case 'edit_user': return adminEditVipCode_(body);
      case 'delete_vip_code': case 'delete_user': return adminDeleteVipCode_(body.code);
      case 'get_vip_codes': case 'get_all_users': return adminGetVipCodes_();
      case 'clear_vip_codes': case 'delete_all_users': return adminClearVipCodes_();
      case 'save_category_overrides': return adminSaveCategoryOverrides_(body.overrides || legacyOverridesFromParams_(body));
      case 'set_hidden_categories': case 'save_view_config': return adminSetHiddenCategories_(Array.isArray(body.hidden) ? body.hidden : parseHidden_(body.hidden_categories || body.hidden || ''));
      case 'set_wallpapers': return adminSetWallpapers_(body.wallpapers, body.interval_min);
      case 'clear_sessions': return adminClearSessions_();
      default: return err_('Unknown admin action: '+action);
    }
  } catch(ex) { return err_(ex.toString()); }
}

function handleVerifyCode_(p, androidCompat) {
  const code=String(p.code||'').trim().toUpperCase();
  if (!code) return androidCompat ? ok_({success:false, message:'الكود فارغ'}) : ok_({valid:false, message:'الكود فارغ'});
  const c=getConfig_(); const codes=getVipCodes_(); const entry=codes[code];
  let valid=!!entry && !entry.disabled;
  if (valid && entry.expires_at) { const t=new Date(entry.expires_at).getTime(); if (!isNaN(t) && Date.now()>t) valid=false; }
  if (!valid) return androidCompat ? ok_({success:false, message:'كود غير صحيح أو منتهي'}) : ok_({valid:false, message:'كود غير صحيح أو منتهي'});
  const deviceId=String(p.device_id||p.device||'').trim();
  if (deviceId) { entry.devices=entry.devices||[]; if (!entry.devices.includes(deviceId)) entry.devices.push(deviceId); codes[code]=entry; saveVipCodes_(codes); }
  const url = entry.playlist_url || c.master_url;
  const payload={
    valid:true, success:true, message:'تم التحقق بنجاح ✅', code, name:entry.name||'مستخدم VIP', expires_at:entry.expires_at||'',
    playlist_url:url, master_url:url, m3u_url:url, server_revision:c.server_revision,
    app_mode:c.app_mode, wallpapers:c.wallpapers||[], wallpaper_interval_min:c.wallpaper_interval_min||5
  };
  return ok_(payload);
}

function handleGetCategories_(p) {
  const c=getConfig_(); if (!c.master_url) return ok_({status:'error', message:'master_url not set', channels:[], categories:[], server_revision:c.server_revision});
  try {
    const res=UrlFetchApp.fetch(c.master_url,{method:'get',muteHttpExceptions:true,followRedirects:true,headers:{'User-Agent':'Mozilla/5.0 (LATCHI IPTV)'}});
    if (res.getResponseCode()!==200) return ok_({status:'error', message:'HTTP '+res.getResponseCode(), channels:[], categories:[], server_revision:c.server_revision});
    const parsed=parseM3U_(res.getContentText('UTF-8'));
    const hidden=(c.hidden_categories||[]).map(x=>String(x).toLowerCase());
    const overrides=c.category_overrides||{};
    const channels=parsed.filter(ch=>!hidden.some(h=>String(ch.group||'').toLowerCase().includes(h))).map(ch=>({
      name:ch.name, url:ch.url, streamUrl:ch.url, logo:ch.logo, logoUrl:ch.logo,
      group:(overrides[ch.group]||{}).name||ch.group, category:(overrides[ch.group]||{}).name||ch.group, group_orig:ch.group, contentType:detectType_(ch)
    }));
    const catMap={}; channels.forEach(ch=>{catMap[ch.group]=(catMap[ch.group]||0)+1;});
    const categories=Object.entries(catMap).map(([name,count])=>({name,count,order:(overrides[name]||{}).order||999})).sort((a,b)=>a.order-b.order);
    return ok_({server_revision:c.server_revision, app_mode:c.app_mode, channels, categories, total:channels.length, wallpapers:c.wallpapers||[], wallpaper_interval_min:c.wallpaper_interval_min||5});
  } catch(ex) { return ok_({status:'error', message:ex.toString(), channels:[], categories:[], server_revision:c.server_revision}); }
}

function handleGetAppUpdate_(p) {
  const c=getConfig_(); const cur=parseInt(p.version_code||'0',10)||0; const isPC=String(p.platform||'').toLowerCase()==='pc';
  const url=isPC ? (c.pc_download_url||'') : (c.apk_download_url||''); const latest=parseInt(c.app_version_code||1,10);
  return ok_({
    update_available: latest>cur && !!url,
    version_code: latest, versionCode: latest,
    version_name: c.app_version_name, versionName: c.app_version_name,
    apk_url: url, apkUrl: url, url,
    force_update: !!c.force_update, forceUpdate: !!c.force_update,
    notes_ar: c.update_notes_ar, notes:{ar:c.update_notes_ar}, releaseNotes:{ar:c.update_notes_ar}
  });
}

  saveSessions_(s); return ok_({status:'ok', server_revision:getConfig_().server_revision});
}

function legacyOverridesFromParams_(p){
  let names={}; let order=[]; try{names=JSON.parse(p.custom_names||'{}');}catch(e){} try{order=JSON.parse(p.custom_order||'[]');}catch(e){}
  const out={}; order.forEach((display,idx)=>{ const original=Object.keys(names).find(k=>names[k]===display)||display; out[original]={name:display, order:idx+1}; });
  Object.keys(names).forEach(k=>{ if(!out[k]) out[k]={name:names[k], order:999}; }); return out;
}
function parseHidden_(v){ if(Array.isArray(v))return v; return String(v||'').split(',').map(x=>x.trim()).filter(Boolean); }
